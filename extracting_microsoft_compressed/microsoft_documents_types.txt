word/
    document.xml             <-- Main document text
    styles.xml               <-- Styles (fonts, headings, etc.)
    numbering.xml            <-- Numbering for lists
    header1.xml, footer1.xml <-- Headers/footers
    media/                   <-- Images, shapes, embedded objects
    theme/                   <-- Document theme
